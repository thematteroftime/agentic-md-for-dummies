# Auto-discovery Registry Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace four hand-curated registries (`forces/__init__.py:FORCE_REGISTRY`, `integrators/__init__.py:INTEGRATOR_REGISTRY`, `tools/lattices/__init__.py:LATTICE_REGISTRY`, `tools/registry.py:_REGISTRY`) plus the `templates/plan_config.schema.json` enum sync and `references/force_types.md` conventions-table sync with a startup-time filesystem scan that writes a `_registry.json` per extension folder. Adding a module becomes "drop file with `REGISTRY_NAME` attribute, restart" — zero registry edits.

**Architecture:** A standalone scanner (`tools/registry/_scanner.py`) walks each of the seven extension folders (`forces/`, `integrators/`, `tools/{analyzers,plotters,aggregators,visualizers,lattices}/`), introspects each `.py` file, and writes per-folder `_registry.json`. The aggregator (`tools/registry.py`) reads all seven manifests at module import, builds the in-memory namespaced + class-name dispatch dicts, and exposes the existing `resolve()` API unchanged. Each per-folder `__init__.py` uses PEP 562 `__getattr__` to expose `from forces import HertzianNonreciprocal` as a lazy registry lookup, so adding a new force class does not require editing `forces/__init__.py`. After scan, the scanner regenerates the schema's `force_type` and `integrator` enums plus the conventions table at the top of `force_types.md`. Migration is hard-cut; public APIs preserved.

**Tech Stack:** Python 3.10, pytest, Taichi 1.7.4 (only relevant for `@ti.data_oriented` class detection in scanner). Standard library only for the scanner — `importlib`, `inspect`, `json`, `pathlib`, `os.path` (mtime checks).

**Source spec:** [`docs/superpowers/specs/2026-05-10-auto-discovery-registry-design.md`](../specs/2026-05-10-auto-discovery-registry-design.md)

```python
# tools/registry/_scanner.py
"""Filesystem scanner that builds per-folder _registry.json manifests.

Public API (callable from the aggregator):
- scan_folder(folder, base_class_dotted) -> manifest dict
- update_registry_json(folder, manifest) -> None
- _folder_dirty(folder) -> bool

Private helpers prefixed with `_`.
"""
from __future__ import annotations
import datetime
import importlib
import inspect
import json
import re
import sys
import traceback
from pathlib import Path
from typing import Optional

VALID_STEM = re.compile(r"^[a-z][a-z0-9_]+$")
SCHEMA_VERSION = 1


def _import_base_class(dotted: Optional[str]):
    """Resolve 'pkg.module:ClassName' to the class object. None means no base."""
    if dotted is None:
        return None
    module_path, class_name = dotted.split(":", 1)
    mod = importlib.import_module(module_path)
    return getattr(mod, class_name)


def _candidate_files(folder: Path) -> list[Path]:
    """Return .py files under folder that aren't excluded by convention."""
    out = []
    for p in sorted(folder.glob("*.py")):
        stem = p.stem
        if stem.startswith("_") or stem == "__init__":
            continue
        out.append(p)
    return out


def _try_import_module(folder: Path, file_path: Path):
    """Import the module corresponding to `<folder.name>.<file.stem>`. Returns
    (module, None) on success or (None, error_dict) on failure."""
    # Compute dotted module path: folder name + file stem
    # Caller must ensure project root is on sys.path.
    pkg_relpath = file_path.relative_to(file_path.parents[0]).with_suffix("")
    # Use the folder's parent to derive the package name; but folder may be
    # nested (tools/lattices/ for example). Simpler: derive from file_path
    # relative to project root.
    from tools.registry._constants import ROOT
    rel = file_path.relative_to(ROOT).with_suffix("")
    dotted = ".".join(rel.parts)
    try:
        # If module was previously imported, reload to pick up file changes.
        if dotted in sys.modules:
            mod = importlib.reload(sys.modules[dotted])
        else:
            mod = importlib.import_module(dotted)
        return mod, None
    except Exception as e:
        return None, {
            "ok": False,
            "registry_name": None,
            "class_name": None,
            "class_path": None,
            "error_summary": f"{type(e).__name__}: {e}",
            "error_traceback_first_line": traceback.format_exc().splitlines()[-3:],
        }


def _find_subclass_or_none(module, base_class) -> tuple[Optional[type], Optional[dict]]:
    """Find the unique public subclass of `base_class` in `module`. Returns
    (cls, None) on success, (None, error_dict) on failure (multiple candidates,
    none, or other anomaly)."""
    if base_class is None:
        # Folders without a base class (analyzers, plotters, ...): pick the
        # unique top-level class whose name does not start with _ and which
        # is defined in this module (not imported).
        candidates = [
            obj for name, obj in inspect.getmembers(module, inspect.isclass)
            if not name.startswith("_") and obj.__module__ == module.__name__
        ]
    else:
        candidates = [
            obj for name, obj in inspect.getmembers(module, inspect.isclass)
            if not name.startswith("_")
            and obj is not base_class
            and obj.__module__ == module.__name__
            and issubclass(obj, base_class)
        ]
    if len(candidates) == 0:
        return None, {
            "ok": False,
            "error_summary": "no public class subclassing the expected base class",
        }
    if len(candidates) > 1:
        return None, {
            "ok": False,
            "error_summary": f"multiple base-class subclasses: {[c.__name__ for c in candidates]}",
        }
    return candidates[0], None


def scan_folder(folder: Path, base_class_dotted: Optional[str]) -> dict:
    """Scan `folder` for .py files, introspect each, return a manifest dict.

    The returned dict has the on-disk _registry.json schema:
      { "_schema_version": 1, "_folder": "<short>", "_base_class": "<dotted>",
        "_scanned_at": "<ISO timestamp>", "_python_version": "...",
        "entries": { <registry_name>: { class_path, registry_name, class_name,
                                        ok, [error_summary], ... } } }
    """
    base_class = _import_base_class(base_class_dotted)
    entries: dict[str, dict] = {}

    for file_path in _candidate_files(folder):
        stem = file_path.stem
        if not VALID_STEM.match(stem):
            entries[stem] = {
                "ok": False,
                "registry_name": None,
                "class_name": None,
                "class_path": None,
                "error_summary": f"file stem '{stem}' violates ^[a-z][a-z0-9_]+$",
            }
            continue

        module, err = _try_import_module(folder, file_path)
        if err is not None:
            entries[stem] = err
            continue

        cls, err = _find_subclass_or_none(module, base_class)
        if err is not None:
            entries[stem] = err
            continue

        # Check REGISTRY_NAME is declared and matches file stem
        registry_name = getattr(cls, "REGISTRY_NAME", None)
        if registry_name is None:
            entries[stem] = {
                "ok": False,
                "class_name": cls.__name__,
                "error_summary": f"class {cls.__name__} missing REGISTRY_NAME class attribute",
            }
            continue
        if registry_name != stem:
            entries[stem] = {
                "ok": False,
                "class_name": cls.__name__,
                "error_summary": (
                    f"class {cls.__name__} declares REGISTRY_NAME='{registry_name}' "
                    f"but file stem is '{stem}' (must match)"
                ),
            }
            continue

        entries[stem] = {
            "ok": True,
            "registry_name": registry_name,
            "class_name": cls.__name__,
            "class_path": f"{module.__name__}:{cls.__name__}",
        }

    return {
        "_schema_version": SCHEMA_VERSION,
        "_folder": folder.name,
        "_base_class": base_class_dotted,
        "_scanned_at": datetime.datetime.now().isoformat(timespec="seconds"),
        "_python_version": ".".join(map(str, sys.version_info[:3])),
        "entries": entries,
    }
```

```python
def update_registry_json(folder: Path, manifest: dict) -> None:

    out_path = folder / "_registry.json"
```

```python
def _folder_dirty(folder: Path) -> bool:
    """Return True if any .py file in `folder` is newer than _registry.json,
    or if the manifest is missing entirely."""
    manifest_path = folder / "_registry.json"
    if not manifest_path.exists():
        return True
    manifest_mtime = manifest_path.stat().st_mtime
    for p in folder.glob("*.py"):
        if p.stat().st_mtime > manifest_mtime:
            return True
    return False
```

```python
# tools/registry/_aggregator.py
"""Aggregator: builds in-memory namespaced + class-name dispatch dicts from
the seven per-folder _registry.json files. Lazy-triggered; mtime-cached.

resolve(name) is the public API. Handles both:
- namespaced lookups: "forces:kalj", "analyzers:pedersen"
- bare class-name lookups (back-compat): "PedersenAnalyzer", "BAOABLangevin"
- bare registry-name lookups: "kalj", "baoab_langevin"
- fully-qualified dotted paths: "forces.kalj:KobAndersenLJ"
"""
from __future__ import annotations
import importlib
import json
from pathlib import Path
from typing import Any, Type

from tools.registry._constants import EXTENSION_FOLDERS
from tools.registry._scanner import scan_folder, update_registry_json, _folder_dirty


# Populated lazily on first resolve() / load_registry() / known() call.
_BY_NAMESPACED: dict[str, str] = {}  # "forces:kalj" -> "forces.kalj:KobAndersenLJ"
_BY_REGISTRY_NAME: dict[str, str] = {}  # "kalj" -> "forces.kalj:KobAndersenLJ"
_BY_CLASS_NAME: dict[str, str] = {}  # "KobAndersenLJ" -> "forces.kalj:KobAndersenLJ"
_FRESH = False


def _ensure_fresh() -> None:
    """Walk all seven extension folders. Rescan any whose .py files are newer
    than their _registry.json. Repopulate the in-memory dicts."""
    global _FRESH
    any_rescanned = False
    for short, folder, base_class in EXTENSION_FOLDERS:
        if not folder.exists():
            continue
        if _folder_dirty(folder):
            manifest = scan_folder(folder, base_class)
            update_registry_json(folder, manifest)
            any_rescanned = True

    # Rebuild in-memory dicts from disk (cheap; ~7 file reads)
    _BY_NAMESPACED.clear()
    _BY_REGISTRY_NAME.clear()
    _BY_CLASS_NAME.clear()
    for short, folder, _ in EXTENSION_FOLDERS:
        manifest_path = folder / "_registry.json"
        if not manifest_path.exists():
            continue
        manifest = json.loads(manifest_path.read_text(encoding="utf-8"))
        for reg_name, entry in manifest.get("entries", {}).items():
            if not entry.get("ok"):
                continue
            cls_path = entry["class_path"]
            ns_key = f"{short}:{reg_name}"
            _BY_NAMESPACED[ns_key] = cls_path
            _BY_REGISTRY_NAME[reg_name] = cls_path
            _BY_CLASS_NAME[entry["class_name"]] = cls_path
    _FRESH = True

    # Schema-enum and conventions-table regeneration land here in Task 21+.


def _import_class(class_path: str) -> type:
    module_path, attr = class_path.split(":", 1)
    mod = importlib.import_module(module_path)
    return getattr(mod, attr)


def resolve(name: str) -> Type[Any]:
    """Resolve a name to a class. Tries namespaced, bare registry name, class
    name (back-compat), and fully-qualified dotted path in that order."""
    if not _FRESH:
        _ensure_fresh()
    # Fully-qualified dotted path
    if "." in name and ":" in name:
        return _import_class(name)
    # Namespaced
    if name in _BY_NAMESPACED:
        return _import_class(_BY_NAMESPACED[name])
    # Class name (back-compat)
    if name in _BY_CLASS_NAME:
        return _import_class(_BY_CLASS_NAME[name])
    # Bare registry name
    if name in _BY_REGISTRY_NAME:
        return _import_class(_BY_REGISTRY_NAME[name])
    raise KeyError(
        f"unknown class '{name}' — register it by dropping a file in the "
        f"matching folder under forces/, integrators/, tools/lattices/, "
        f"tools/analyzers/, tools/plotters/, tools/aggregators/, or "
        f"tools/visualizers/. Known: namespaced={sorted(_BY_NAMESPACED)[:5]}..., "
        f"by-class={sorted(_BY_CLASS_NAME)[:5]}..."
    )


def known() -> list[str]:
    """List all registered names (namespaced form)."""
    if not _FRESH:
        _ensure_fresh()
    return sorted(_BY_NAMESPACED)


def register(name: str, target: str) -> None:
    """Runtime registration hook (preserves legacy API). Test-only — production
    code should drop a file in the appropriate folder instead."""
    if not _FRESH:
        _ensure_fresh()
    _BY_NAMESPACED[name] = target
    if ":" in target:
        _BY_REGISTRY_NAME[name.split(":", 1)[-1]] = target
        _BY_CLASS_NAME[target.rsplit(":", 1)[-1]] = target


def load_registry(short_name: str) -> dict[str, type]:
    """Return the per-folder dict {registry_name: class}. Used by per-folder
    __init__.py to populate FORCE_REGISTRY / INTEGRATOR_REGISTRY / etc."""
    if not _FRESH:
        _ensure_fresh()
    out: dict[str, type] = {}
    for short, folder, _ in EXTENSION_FOLDERS:
        if short != short_name:
            continue
        manifest_path = folder / "_registry.json"
        if not manifest_path.exists():
            continue
        manifest = json.loads(manifest_path.read_text(encoding="utf-8"))
        for reg_name, entry in manifest.get("entries", {}).items():
            if entry.get("ok"):
                out[reg_name] = _import_class(entry["class_path"])
        return out
    raise KeyError(f"unknown extension folder short name '{short_name}'")


def get_class_by_class_name(short_name: str, class_name: str) -> Type[Any] | None:
    """For PEP 562 __getattr__ in per-folder __init__.py: find a class by its
    Python class name (not registry_name) within a single folder. Returns None
    if not found."""
    if not _FRESH:
        _ensure_fresh()
    for short, folder, _ in EXTENSION_FOLDERS:
        if short != short_name:
            continue
        manifest_path = folder / "_registry.json"
        if not manifest_path.exists():
            return None
        manifest = json.loads(manifest_path.read_text(encoding="utf-8"))
        for entry in manifest.get("entries", {}).values():
            if entry.get("ok") and entry.get("class_name") == class_name:
                return _import_class(entry["class_path"])
        return None
    return None
```

```python
def update_conventions_table(md_path: Path, rows: list[dict]) -> None:
    """Rewrite ONLY the conventions-table block in `references/force_types.md`.

    The table is the first markdown table after a `## Conventions table`
    heading. Everything before that heading and everything after the first
    blank line following the table is preserved verbatim.

    `rows` is a list of dicts with keys: force_type, n_means, default_ic, ndim, units_regime.
    """
    text = md_path.read_text(encoding="utf-8")
    lines = text.splitlines(keepends=True)

    # Find the heading
    try:
        heading_idx = next(i for i, line in enumerate(lines)
                           if line.strip() == "## Conventions table")
    except StopIteration:
        raise RuntimeError(f"'## Conventions table' heading not found in {md_path}")

    # Find table start: the first line starting with "|" after the heading
    i = heading_idx + 1
    while i < len(lines) and not lines[i].lstrip().startswith("|"):
        i += 1
    table_start = i

    # Find table end: first blank line (or end-of-file) after the table starts
    while i < len(lines) and lines[i].strip().startswith("|"):
        i += 1
    table_end = i

    # Render new table
    new_table = ["| force_type | `N` means | default IC | ndim | units_regime |\n",
                 "|------------|-----------|-----------|------|--------------|\n"]
    for r in rows:
        new_table.append(
            f"| `{r['force_type']}` | {r['n_means']} | {r['default_ic']} | "
            f"{r['ndim']} | `{r['units_regime']}` |\n"
        )

    # Splice
    new_lines = lines[:table_start] + new_table + lines[table_end:]
    md_path.write_text("".join(new_lines), encoding="utf-8")
```

```python
#!/usr/bin/env python3
"""scan_registry.py — manually trigger a registry rescan or check freshness.

Modes:
  (no flag)      Force a full rescan and write all _registry.json files.
                 Also rewrites schema enums and force_types.md conventions table.
  --check        Dry-run mode: scan in memory, compare to disk, exit non-zero
                 if any folder would change. Used in CI / pre-commit hooks.
  --list         Print all registered names, grouped by folder.

Exit codes:
  0  OK
  1  drift detected (only in --check)
  2  unexpected error
"""
from __future__ import annotations
import argparse
import json
import sys
from pathlib import Path

# Add project root to sys.path so `import tools.registry...` works
_ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(_ROOT))


def main() -> int:
    parser = argparse.ArgumentParser(description=__doc__,
                                     formatter_class=argparse.RawDescriptionHelpFormatter)
    parser.add_argument("--check", action="store_true",
                        help="dry-run; exit 1 on drift")
    parser.add_argument("--list", action="store_true",
                        help="print all registered names")
    args = parser.parse_args()

    from tools.registry._scanner import scan_folder
    from tools.registry._constants import EXTENSION_FOLDERS

    if args.list:
        for short, folder, base in EXTENSION_FOLDERS:
            manifest_path = folder / "_registry.json"
            if not manifest_path.exists():
                print(f"  {short}: <no manifest>")
                continue
            entries = json.loads(manifest_path.read_text(encoding="utf-8")).get("entries", {})
            ok = sorted(k for k, e in entries.items() if e.get("ok"))
            broken = sorted(k for k, e in entries.items() if not e.get("ok"))
            print(f"  {short}: {ok}" + (f"  broken: {broken}" if broken else ""))
        return 0

    if args.check:
        drift: list[str] = []
        for short, folder, base in EXTENSION_FOLDERS:
            manifest_path = folder / "_registry.json"
            if not manifest_path.exists():
                drift.append(f"{short}: no manifest on disk")
                continue
            on_disk = json.loads(manifest_path.read_text(encoding="utf-8"))
            fresh = scan_folder(folder, base)
            if on_disk.get("entries", {}) != fresh.get("entries", {}):
                drift.append(f"{short}: drift detected")
        if drift:
            print("DRIFT:", ", ".join(drift), file=sys.stderr)
            return 1
        print("OK: all _registry.json files are in sync with disk")
        return 0

    # Full rescan mode
    from tools.registry._aggregator import _ensure_fresh
    # Touch one file in each folder to force rescan
    for _, folder, _ in EXTENSION_FOLDERS:
        for p in folder.glob("*.py"):
            p.touch()
            break
    _ensure_fresh()
    print("Rescanned all seven folders + regenerated schema enums + conventions table")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
